3   import java.util.ArrayList;
4   import java.util.LinkedList;
5   import varivaltaus.varivaltaus.kayttoliittyma.*;
6   
7   /**
8    * Käynnistää ja pyörittää pelisession alusta loppuun.
9    *
10   * @author juma
11   */
12  public class Pelinpyorittaja {
13  
14      private final Ruudukko ruudukko;
15      private final LinkedList<Pelaaja> pelaajat;
16      private final Kayttoliittyma ui;
17      private final int variLkm;
18  
19      /**
20       * Tallentaa pelin pyörittämiseen tarvittavat olioviitteet.
21       *
22       * @param ruudukko Pelisessiossa käytettävä ruudukko.
23       * @param pelaajat Pelisession pelaajat.
24       * @param ui Pelisession käyttöliittymä.
25       */
26      public Pelinpyorittaja(Ruudukko ruudukko, LinkedList<Pelaaja> pelaajat, Kayttoliittyma ui) {
27          this.ruudukko = ruudukko;
28          this.variLkm = ruudukko.getVariLkm();
29          this.pelaajat = pelaajat;
30          this.ui = ui;
31      }
32  
33      /**
34       * Käynnistää pelinpyörityksen. Kysyy käyttöliittymältä pelaajien uusia
35       * värejä pelaajilta vuoronperään sekä päivittää pelilaudan sisällön ja
36       * pyytää käyttöliittymää päivittämään pelilautanäkymän, lopuksi pyytää
37       * käyttöliittymää julistamaan voittajan.
38       */
39      public void aloitaPeli() {
40          valtaaPelaajilleAloitusAlueet();
41          this.ui.paivitaPelilauta();
42  
43          Pelaaja voittaja = null;
44  
45          while (voittaja == null) {
46              annaSeuraavaVuoro();
47  
48              if (this.voittaako(this.pelaajat.peekLast())) {
49                  voittaja = this.pelaajat.getLast();
50              }
51              
52              this.ui.paivitaPelilauta();
53          }
54  
55          this.ui.julistaVoittaja(voittaja);
56      }
57  
58      private void valtaaPelaajilleAloitusAlueet() {
59          for (Pelaaja p : this.pelaajat) {
60              this.valtaaUudetRuudut(p);
61          }
62      }
63  
64      private boolean voittaako(Pelaaja p) {
65          return p.getAlueenKoko() >= (this.ruudukko.getRuutujenMaara() + 1) / 2;
66      }
67  
68      private void annaSeuraavaVuoro() {
69          Pelaaja seuraava = this.pelaajat.removeFirst();
70          this.pelaajat.addLast(seuraava);
71          pelaaVuoro(seuraava);
72      }
73  
74      private void pelaaVuoro(Pelaaja p) {
75          int vari = this.ui.kysyVari(vapaatVarit(), p);
76          
77          p.vaihdaVari(vari);
78          this.valtaaUudetRuudut(p);
79      }
80  
81      private ArrayList<Integer> vapaatVarit() {
82          ArrayList<Integer> varit = new ArrayList();
83  
84          for (int i = 1; i <= variLkm; i++) {
85              if (onkoVapaaVari(i)) {
86                  varit.add(i);
87              }
88          }
89  
90          return varit;
91      }
92  
93      private boolean onkoVapaaVari(int i) {
94          for (Pelaaja p : this.pelaajat) {
95              if (p.getNykyinenVari() == i) {
96                  return false;
97              }
98          }
99  
100         return true;
101     }
102 
103     private void valtaaUudetRuudut(Pelaaja p) {
104         ArrayList<Ruutu> alue = p.getListaAlueesta();
105 
106         for (Ruutu r : alue) {
107             valtaaYmparoivatRuudut(r, p);
108         }
109 
110     }
111 
112     private void valtaaYmparoivatRuudut(Ruutu r, Pelaaja p) {
113 
114         ArrayList<Ruutu> viereiset = this.ruudukko.getViereisetRuudut(r.getX(), r.getY());
115 
116         for (Ruutu v : viereiset) {
117             if (!v.isVallattu() && v.getVari() == p.getNykyinenVari()) {
118                 p.valtaaRuutu(v);
119                 valtaaYmparoivatRuudut(v, p);
120             }
121         }
122 
123     }
124 }
