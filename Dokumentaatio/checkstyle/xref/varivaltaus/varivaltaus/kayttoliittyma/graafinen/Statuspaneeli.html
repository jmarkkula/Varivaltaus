1   package varivaltaus.varivaltaus.kayttoliittyma.graafinen;
2   
3   import java.awt.GridLayout;
4   import java.util.HashMap;
5   import java.util.LinkedList;
6   import javax.swing.JPanel;
7   import varivaltaus.varivaltaus.pelilogiikka.Pelaaja;
8   import varivaltaus.varivaltaus.pelilogiikka.Ruudukko;
9   
10  /**
11   * Sisältää pelaajapaneelit, jotka näyttävät käyttäjälle ilmoituksia
12   * pelitilanteesta.
13   */
14  public class Statuspaneeli extends JPanel {
15  
16      private final HashMap<Pelaaja, Pelaajapaneeli> pelaajapaneelit;
17  
18      protected Statuspaneeli(LinkedList<Pelaaja> p, Varit v, Ruudukko r) {
19          this.setLayout(new GridLayout(1, p.size()));
20  
21          this.pelaajapaneelit = new HashMap<>();
22  
23          for (Pelaaja pelaaja : p) {
24              Pelaajapaneeli uusi = new Pelaajapaneeli(pelaaja, v, r);
25              this.pelaajapaneelit.put(pelaaja, uusi);
26              this.add(uusi);
27          }
28      }
29  
30      protected void paivita() {
31          for (Pelaajapaneeli p : this.pelaajapaneelit.values()) {
32              p.paivita();
33          }
34      }
35  
36      protected void asetaTeksti(Pelaaja kenelleAsetetaan, String teksti) {
37          this.pelaajapaneelit.get(kenelleAsetetaan).asetaTeksti(teksti);
38      }
39  }
