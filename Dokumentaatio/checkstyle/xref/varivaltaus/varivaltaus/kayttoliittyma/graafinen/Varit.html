1   package varivaltaus.varivaltaus.kayttoliittyma.graafinen;
2   
3   import java.awt.Color;
4   import java.util.HashMap;
5   
6   /**
7    * Hallinnoi pelilogiikan värejä edustavien numeroiden mäppäystä varsinaisiksi
8    * Color-olioiksi.
9    */
10  public class Varit {
11  
12      HashMap<Integer, Color> varit;
13  
14      public Varit() {
15          this.varit = new HashMap<>();
16          this.luoVarit();
17      }
18  
19      public Color getVari(int varinNumero) {
20          if (!this.varit.containsKey(varinNumero)) {
21              throw (new Error("Värejä ei ole luotu noin montaa Varit-luokassa"));
22          }
23  
24          return this.varit.get(varinNumero);
25      }
26  
27      private void luoVarit() {
28          this.varit.put(1, new Color(175, 230, 173)); //v.vihreä
29          this.varit.put(2, new Color(203, 147, 230)); //lila
30          this.varit.put(3, new Color(230, 163, 167)); //v.pun.
31          this.varit.put(4, new Color(232, 255, 154)); //v.kelt.
32          this.varit.put(5, new Color(146, 171, 229)); //v.sin
33          this.varit.put(6, Color.MAGENTA);
34          this.varit.put(7, Color.blue);
35          this.varit.put(8, Color.GREEN);
36  
37      }
38  
39  }
