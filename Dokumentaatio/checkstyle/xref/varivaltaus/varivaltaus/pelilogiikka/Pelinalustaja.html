3   import java.util.LinkedList;
4   
5   /**
6    * Luokka alustaa ja säilyttää pelissä käytettävän peliruudukon ja pelaajat.
7    *
8    */
9   public class Pelinalustaja {
10  
11      private final Ruudukko ruudukko;
12      private LinkedList<Pelaaja> pelaajat;
13  
14      /**
15       * Konstruktori generoi ruudukon, luo pelaajat ja asettaa pelaajat
16       * aloitusalueilleen.
17       *
18       * @param leveys Ruudukon leveys.
19       * @param korkeus Ruudukon korkeus.
20       * @param variLkm Peliin haluttu eri värien määrä.
21       */
22      public Pelinalustaja(int leveys, int korkeus, int variLkm) {
23          this.ruudukko = new Ruudukko(leveys, korkeus, variLkm);
24          this.luoPelaajat();
25      }
26  
27      /**
28       * Palauttaa viitteen pelin ruudukkoon.
29       *
30       * @return Viite Ruudukkoon.
31       */
32      public Ruudukko getRuudukko() {
33          return this.ruudukko;
34      }
35  
36      /**
37       * Palauttaa listan pelin pelaaja-olioista.
38       *
39       * @return Pelaajat LinkedListissä.
40       */
41      public LinkedList<Pelaaja> getPelaajat() {
42          return this.pelaajat;
43      }
44  
45      private void luoPelaajat() {
46          this.pelaajat = new LinkedList<>();
47  
48          Pelaaja p1 = luoPelaaja(1, this.ruudukko.getRuutu(0, 0));
49          Pelaaja p2 = luoPelaaja(2, this.ruudukko.getRuutu(this.ruudukko.getLeveys() - 1, this.ruudukko.getKorkeus() - 1));
50  
51          this.pelaajat.add(p1);
52          this.pelaajat.add(p2);
53      }
54  
55      private Pelaaja luoPelaaja(int pelaajaNro, Ruutu aloitusruutu) {
56  
57          Pelaaja p = new Pelaaja(pelaajaNro, aloitusruutu.getVari());
58          p.valtaaRuutu(aloitusruutu);
59  
60          return p;
61      }
62  
63  }
